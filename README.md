# Installation
```
npm install anyback-react
```

# Getting started
```javascript

import { AnyBackAdminPanel } from 'anyback-react';

ReactDOM.createRoot(
  document.querySelector('#root') // or other
)
.render(
  <AnyBackAdminPanel
    options={{
        // ...your options
    }}
  />
)

```
